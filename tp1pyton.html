Cette exercice consiste a resoudre un systeme d'equation lineare avec la matrice inverse

    import numpy as np
    import math 
    print("Ce programme consiste à remplir une matrice et un vecteur")
    
    n = int(input("Entrez l'ordre de la matrice : "))
    
    # Créez la matrice A en demandant à l'utilisateur d'entrer les éléments
    A = np.zeros((n, n))
    print("Entrez les éléments de la matrice A :")
    for i in range(n):
        for j in range(n):
            A[i, j] = float(input("A["+str(i)+"]["+str(i)+"]="))
    
    # Créez le vecteur y en demandant à l'utilisateur d'entrer les éléments
    y = np.zeros(n)
    print("Entrez les éléments du vecteur y :")
    for i in range(n):
        y[i] = float(input("A["+str(i)+"]["+str(i)+"]= "))
    
    # Méthode 1
    x1=np.zeros(n)
    x1 = np.linalg.inv(A) @ y
    print("x1 =", x1)
    
    # Méthode 2
    x2=np.zeros(n)
    x2 = np.linalg.solve(A, y)
    print("x2 =", x2)
